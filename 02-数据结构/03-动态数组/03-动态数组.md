## 2 代码实现

```ts
class myClass {
	capacity: number;
	size: number;
	data: any[];
  constructor(capacity: number = 10) {
		// 数组开辟的空间容量
    this.capacity = capacity;
		// 数组实际的成员个数; 下一个元素成员 即将会插入的位置
		this.size = 0
    this.data = new Array(capacity);
  }
	// 获取数组的实际成员个数
	getsize() {
		return this.size
	}
	// 获取当前数组的长度
	getCapacity() {
		return this.data.length
	}
	// 返回数组是否为空
  isEmpty(){
    return this.size === 0
	}

	//向所有元素后添加一个新元素
  addLast(e: any) {
    this.add(this.size, e);
  }
  // 在所有元素前添加一个新元素
  addFirst(e: any) {
    this.add(0, e);
  }
  // 在第index个位置插入一个新元素e
  add(index: number, e: any) {
    if (this.size === this.data.length) {
      throw new Error("Add failed. Array is full.");
    }
    if (index < 0 || index > this.size) {
      throw new Error("Add failed. Require index >= 0 && index <= this.size");
    }
    // 向后拷贝/平移覆盖
    for (let i = this.size - 1; i >= index; i--) {
      this.data[i + 1] = this.data[i];
    }
  }

}
```
